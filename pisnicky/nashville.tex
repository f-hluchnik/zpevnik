\vers{1}{
Kdo \chord{A}umí zahrát gé á há cé, od přírody \chord{E}touží\\
aspoň jednou \chord{D}podívat se \chord{E}za pořádnou \chord{A}louži.
}

\refrain{
\chord{A}Hej, Mary Lou, sejdem' se v Nashvil\chord{D}lu,\\

\vers{3}{

dělá skvělý myšlenky a projde každým oudem.
} \refsm{}

\vers{5}{
Je tu houslák Jimmy a je tu basák Ben,\\
prostě celá squadra samejch slavnejch jmen.
} \refsm{}

\vers{6}{

\vers{7}{
A pak přichází Mary Lou a učí nás svý tance\\
a na břehu řeky Cumberland si zalamujem palce.
} \refsm{}

\vers{8}{
