\hyt{jajedudal}
\song{Já jedu dál} \interpret{tucny}{Greenhorns}

\vers{1}{
	Zas \chord{D}modrý nebe mám nad hlavou a \chord{D\7}dobře znám silnici toulavou, já jedu \chord{G}dál,\\
	jak jsem si \chord{D}přál a \chord{A}se starou kytarou prašnejch cest já jsem zas \chord{D}král.
}

\vers{2}{
	Ty máš ráda lesky blatníků a já zas samotu patníku, tak jedu dál,\\
	jak jsem si přál, a se starou kytarou prašnejch cest já jsem zas král.
}

\vers{3}{
	Mý nebe je nízký pro tvý oko, jsi zvyklá létat moc vysoko, já jedu dál,\\
	jak jsem si přál, a se starou kytarou prašnejch cest já jsem zas král.
}

\vers{4}{
	Má láska pro tebe nic není, tvý srdce je prázdný stavení, já jedu dál,\\
	jak jsem si přál, a se starou kytarou prašnejch cest já jsem zas král.
}

\vers{5}{
	Teď bez tebe mi asi bude líp, já stopnu si Tatru nebo džíp a jedu dál,\\
	jak jsem si přál, a se starou kytarou prašnejch cest já jsem zas král.
}

\vers{6}{
	Zas modrý nebe mám nad hlavou a dobře znám silnici toulavou, já jedu dál,\\
	jak jsem si přál, a se starou kytarou prašnejch cest já jsem zas král.
}
