\hyt{buchetjespousta}
\song{Buchet je spousta}
\sub{z pohádky Kouzla králů}
\ns

\vers{1}{
\chord{C}Z pekárny \chord{Am}světa kraj za \chord{F}láskou \chord{G}svou,\\
\chord{C}nejednou \chord{F}překonal jsem \chord{Am}vůli \chord{G}zlou,\\
\chord{Dm}a moje \chord{G}princezna, \chord{Em}ta jistě \chord{Am}rozezná,\\
\chord{F}že ji mám \chord{G}vážně \chord{C}rád.
}

\vers{2}{
A v jeho blízkosti se náhle zdá,\\
že létat v oblacích se málem dá.\\
Dát ruku na ruku, netřeba záruku,\\
jeho se přece nelze vůbec bát!
}

\refrainn{1}{
\chord{C}Oslava veliká ať \chord{F}už se chys\chord{G}tá,\\
\chord{Em}buchet je \chord{Am}spousta, \chord{Dm}lahodná \chord{G}sousta.\\
\chord{C}Ať naše láska je vždy \chord{F}stejně čis\chord{G}tá\\
\chord{F}a pevná jako \chord{G}brány \chord{C}trám.
}

\vers{3}{
Hezká jak obrázek a zábavná,\\
a pýchu princezen tak málo zná.\\
Vyrostla v dostatku, jak holka ze statku,\\
umí za práci brát.
}

\vers{4}{
Konečně kluk, co o mně zájem má,\\
přitažlivost je jistě vzájemná.\\
A srdce z rubínu má barvu po vínu,\\
starý král novému vždy má jej dát.
}

\refrainn{2}{
Ta naše dvojice je překvapivá,\\
a dle všech známek, pekárna -- zámek\\
je trasa, která někdy krátce trvá,\\
a já ji dobře, lásko, znám.
}
