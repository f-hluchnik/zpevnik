\hyt{hudsonskysify}
\song{Hudsonský šífy}
\interpret{wabidanek}{Wabi Daněk}

\vers{1}{
Ten, kdo \chord{Am}nezná hukot vody lopat\chord{C}kama vířený, jako \chord{G}já, jó jako \chord{Am}já,\\
kdo Hud\chord{Am}sonský slapy nezná sírou \chord{G}pekla sířený,\\
ať se \chord{Am}na Hudsonský \chord{G}šífy najmout \chord{Am}dá.\mm \chord{G}jo-ho-\chord{Am}ho
}

\vers{2}{
Ten, kdo nepřekládal uhlí, šíf když na mělčinu vjel, málo zná, málo zná.\\
Ten, kdo neměl tělo ztuhlý, až se nočním chladem chvěl,\\
ať se na Hudsonský šífy najmout dá. jo-ho-ho
}

\refrain{
\chord{F}Ahoj, páru tam \chord{Am}hoď, ať \chord{G}do pekla se dříve dohra\chord{Am}bem! \chord{G}jo-ho-\chord{Am}ho,\mm \chord{G}jo-ho-\chord{Am}ho
}

\vers{3}{
Ten, kdo nezná noční zpěvy zarostenejch lodníků, jako já, jó jako já,\\
ten, kdo cejtí se bejt chlapem, umí dělat rotyku,\\
ať se na Hudsonský šífy najmout dá. jo-ho-ho
}

\vers{4}{
Ten, kdo má na bradě mlíko, kdo se rumem neopil, málo zná, málo zná.\\
Kdo necejtil hrůzu z vody, kde se málem utopil,\\
ať se na Hudsonský šífy najmout dá. jo-ho-ho
} \refsm{}

\vers{5}{
Kdo má roztrhaný boty, kdo má pořád jenom hlad, jako já, jó jako já,\\
kdo chce celý noci čuchat pekelnýho vohně smrad,\\
ať se na Hudsonský šífy najmout dá. jo-ho-ho
}

\vers{6}{
Kdo chce zhebnout třeba zítra, komu je to všechno fuk, kdo je sám, jó jako já,\\
kdo má srdce v správným místě, kdo je prostě príma kluk,\\
ať se na Hudsonský šífy najmout dá. jo-ho-ho
} \refsm{}
