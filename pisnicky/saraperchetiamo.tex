\hyt{saraperchetiamo}
\song{Sar\`{a} perché ti amo} \interpret{trikyapovery}{Ricchi e Poveri}
\hyl{hejholazpivejsisam}{Hej, hola, zpívej si sám}
\hyl{laskajevytah}{Láska je výtah}

\vers{1}{
\chord{E}Che confusione, sar\`{a} perché ti amo, \chord{C\kk m}\`{e} un emozione, che cresce piano piano,\\
\chord{A}stringimi forte e stammi pi\`{u} vicino, \chord{H}se ci sto bene, sar\`{a} perché ti amo.
}

\vers{2}{
Io canto al ritmo del dolce tuo respiro, \`{e} primavera, sar\`{a} perché ti amo,\\
cade una stella, ma dimmi dove siamo, che te ne frega, sar\`{a} perché ti amo.
}

\refrain{
\textbackslash:\chord{E}E vola vola si sa, sempre pi\`{u} in alto si va,\\
\chord{C\kk m}e vola vola con me, il mondo \`{e} matto perché\\
\chord{A}e se l'amore non c'\`{e}, basta una sola can\chord{H}zone,\\
per far confusione, fuori e dentro di te.:\textbackslash
}

\vers{3}{
Ma dopo tutto, che cosa c'\`{e} di strano, \`{e} una canzone, sar\`{a} perché ti amo,\\
se cade il modno, allora co spostiamo, se cade il mondo, sar\`{a} perché ti amo.
}

\vers{4}{
Stringimi forte e stammi pi\`{u} vicino, e cosi bello che non mi sembra vero,\\
se il mondo \`{e}matto che cosa c'\`{e} di strano, matto per matto, almeno noi ci amiamo.
} \refsm{}
