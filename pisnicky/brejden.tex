\hyt{brejden}
\song{Brej den}
\interpret{plihal}{Karel Plíhal}

\vers{1}{
Brej \chord{C\maj}den, pane \chord{F\maj}slimáku,\\
brej \chord{C\maj}den, jakpak \chord{F\maj}se vám klouže \chord{G}po slině?\\
Já \chord{C\maj}vím, máte mě \chord{F\maj}na háku,\\
ale \chord{G}já vás zcela chápu, tak to nesu hrdin\chord{C}ně.
}

\vers{2}{
Brej den, paní žížalo,\\
brej den, tak co, půjdete dnes na ryby?\\
Já vím, vás to dožralo,\\
ale já vás do tý piksle nedal, já mám alibi.
}

\vers{3}{
Brej den, paní slonice,\\
brej den, můžuj vám vzít míru na župan?\\
Já vím, v černé kronice,\\
budou zítra psát, že jsem byl k smrti udupán.
}

\vers{4}{
Brej den, paní správcová,\\
brej den, pomůžu vám s prádlem na půdu,\\
brej den, proč jste taková?\\
Tam, kam mě teď posíláte, tam já nepudu.
}
