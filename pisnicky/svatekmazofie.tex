\hyt{svatekmazofie}
\song{Svátek má Žofie}
\interpret{navarova}{Zuzana Navarová}
\ns

\vers{1}{
\chord{Em}Za oknem fuchsie, tak \chord{Am}bičav mange \chord{H\7}fotka\footnote{údajně \uv{pošli mi fotku}, je však možné, že jde o neologismus se zvukomalebnou funkcí},\\
\chord{Am}že noci \chord{C}upije ten, \chord{H\7}kdo tě jednou potká.\\
Za oknem fuchsie a hromská práce k tomu,\\
že noci upije ten, kdo tě vezme domů.
}

\vers{2}{
Za oknem fuchsie, tak bičav mange fotka,\\
že noci upije, že luna vlasy protká.\\
Za oknem fuchsie a jedna píseň krotká,\\
sec hadry mordie\footnote{z fr. mort de Dieu, smrt Bohu} a do džusu je vodka.
}
\ns

\bridge{
\chord{Am}Kdo nejed' mango, smutně \chord{H}žil, \chord{Am}pro toho tango, z plna \chord{H}žil.
}

\vers{3}{
Za oknem fuchsie, tak bičav mange fotka,\\
že noci upije ten, kdo tě jednou potká.\\
Za oknem fuchsie a pod kamenem potkan,\\
sec hadry mordie a na pět klení vodka.
} \bridge{}
\ns

\refrain{
Ať nikdy \chord{Em}nezkalí se, \chord{H\7}děj se vůle, \chord{Em}stříbro s \chord{E\7}cínem,\\
\chord{Am}ať příští \chord{D\7}jitra tvá jsou \chord{G}jasná,\chord{H\7}\\
tak jako \chord{Em}slunce na vo\chord{H\7}dě a jak ta \chord{Em}voda s \chord{E\7}vínem,\\
\chord{Am}jak hvězdy, \chord{D\7}co mi choděj' \chord{H\7}do sna.
}

\vers{4}{
Za oknem fuchsie, tak bičav mange fotka,\\
nemáš-li, kup si je, až luna vlasy protká.\\
Za oknem fuchsie a jedna píseň krotká,\\
svátek má Žofie, a ta má v těle šotka.
}
