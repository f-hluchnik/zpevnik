\song{Jdou po mně, jdou}
\ns

\vers{1}{
Býval jsem \chord{D}chudý, jak \chord{G}kostelní \chord{D}myš,\\
na půdě \chord{F\kk m}půdy jsem \chord{Hm}míval svou \chord{A}skrýš.\\
\rep{\chord{G}Pak jednou v \chord{D}létě, \chord{A}řek' jsem si: \chord{Hm}bať,\\
\chord{G}svět facku\chord{D}je tě, a \chord{A}tak mu to \chord{D}vrať.}
}

\vers{2}{
Když mi dát nechceš, já vezmu si sám,\\
zámek jde lehce a adresu znám.\\
\rep{Zlato jak zlato, dolar, či frank,\\
tak jsem šel na to do National Bank.}
}

\refrainn{1}{
Jdou po mně, jdou, jdou, jdou,\\
na každém rohu mají fotku mou,\\
kdyby mě chytli, jó, byl by ring,\\
\chord{G}tma jako v \chord{D}pytli je v \chord{C}celách Sing \chord{G}Sing. \chord{A}jé, \chord{D}jé
}

\vers{3}{
Ve státě Iowa byl od poldů klid,\\
chudinká vdova mi nabídla byt.\\
\rep{Byla to kráska, já měl peníze,\\
tak začla láska jak z televize.}
}

\vers{4}{
Však půl roku na to řekla mi: Dost.\\
Tobě došlo zlato, mně trpělivost.\\
\rep{Sbal svých pár švestek a běž si, kam chceš.\\
Tak jsem na cestě a chudý, jak veš.}
}\refsm{1}

\vers{5}{
Teď ve státě Utahžiju spokojen,\\
pípu jsem utáh' a straním se žen.\\
\rep{Kladou mi pasti a do pastí špek,\\
já na ně mastím, jen ať mají vztek.}
}
