\hyt{zlataklec}
\song{Zlatá klec} \interpret{spiritualkvintet}{Spirituál kvintet}

\vers{1}{
Tvůj \chord{A}stín \chord{A\sus{4}}mne dopro\chord{A}vází,\chord{A\sus{4}} mám \chord{A}koní\chord{A\+} sedm \chord{D\add{6}}párů, \chord{E\7}\\
\chord{A}síň \chord{A\maj}vzácných \chord{A\7}darů, však \chord{Hm}láska \chord{Dm}tvá mi \chord{E\7}schází.
}

\vers{2}{
Mám všechno, co se mi zlíbí, mám kočár, svůj dům s parkem,\\
víš, s každým dárkem tvá láska víc mi chybí.
}

\vers{3}{
Mám květů plné vázy, mám vzápětí, oč stojím,\\
krásně se strojím, však láska tvá mi schází.
}

\vers{4}{
Tak málo ze snů zbývá, můj svět nádherných věcí\\
stává se klecí, kde slavík nezazpívá.
}

\vers{5}{
Chlad nocí střídá denní, i vzpomínky mne zebou,\\
toužím být s tebou, však láska tvá tu není.
}

\vers{6}{
Tvůj stín mne doprovází, mám koní sedm párů,\\
síň vzácných darů, však láska tvá mi schází.
}
