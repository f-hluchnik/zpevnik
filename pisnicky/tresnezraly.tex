\hyt{tresnezraly}
\song{Třešně zrály}
\vers{1}{
\chord{C}Jó, třešně zrály, \chord{G\7}zrovna třešně \chord{C}zrály,\\
sladký \chord{E\7}třešně \chord{Am}zrály a \chord{F}teplej \chord{G\7}vítr \chord{C}vál.\\
A já \chord{G\7}k horám \chord{Am}v dáli, k těm \chord{Dm}modrejm \chord{G\7}horám \chord{C}v dáli,\\
sluncem, \chord{E\7}který\chord{Am} \sm pá\chord{F}lí, tou \chord{Dm}dobou \chord{G\7}stádo \chord{C}hnal.
}

\refrain{
\chord{C}Jó, \chord{G\7}třešně \chord{Am}zrály, \chord{Dm}zrovna \chord{G\7}třešně \chord{C}zrály, sladký \chord{E\7}třešně \chord{Am}zrály a \chord{Dm\7}jak to \chord{G\7}bylo \chord{C}dál?
}

\vers{2}{
Tam, jak je ta skála, ta velká bílá skála,\\
tak tam vám holka stála a bourák opodál.\\
A moc se na mě smála, z dálky už se smála,\\
i z blízka se pak smála a já se taky smál.
}\refsm{}

\vers{3}{
Řekla, že už dlouho mě má ráda, dlouho mě má ráda,\\
dávno mě má ráda, prej, abych si ji vzal.\\
Ať nechám ty svý stáda, že léta pilně střádá,\\
abych ji měl rád a žil s ní jako král.
}\refsm{}

\vers{4}{
Pokud je mi známo, já řek' jenom: Dámo,\\
milá hezká dámo, zač bych potom stál?\\
Ty můj typ nejsi, já mám svoji Gracy,\\
svoji malou Gracy, a tý jsem srdce dal.
}\refsm{}

\vers{5}{
Tam u tý skály dál třešně zrály,\\
sladký třešně zrály a vlahej vítr vál.\\
A já k horám v dáli, k těm modrejm horám v dáli\\
sluncem, který pálí, jsem hnal svý stádo dál.
}
