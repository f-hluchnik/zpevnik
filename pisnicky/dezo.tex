\hyt{dezo}
\song{Dežo} \interpret{nohavica}{Jaromír Nohavica}

\nv\textbf{Am Am\bas{G} F E}

\vers{1}{
Dežo má nože a já panebože ani tyč.\\
Říkám si \uv{Du bist ein Esel},\\
kam jsi to vlezl, jdi pryč.\\
Temný kout tohohle města,\\
to mě mohlo napadnout,\\
že mě tu někdo ztrestá, vole,\\
tady je minové pole.
}

\vers{2}{
Dežo je borec, sune se jak ledoborec tup tudu.\\
A já, mameluk,\\
se pivem namelu, a pak tu du.\\
Temný kout, milá slečno,\\
to mě mohlo napadnout,\\
že je tu nebezpečno, slova, slova, slova.\\
Hlavně neprovokovat.
}

\vers{3}{
Mý vlasy zrezlý mi za ty roky slezly, mám pleš.\\
Dežo se šine, říká mi, skine, kam jdeš?\\
Temný kout, má pleš se leskne,\\
to mě mohlo napadnout,\\
že mě tu někdo pleskne bubu, bubu, bubu,\\
přes tu moji bílou hubu.
} \refsm{}
