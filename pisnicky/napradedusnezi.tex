\hyt{napradedusnezi}
\song{Na Pradědu sněží} \interpret{plihal}{Karel Plíhal}

\note{originál od H2}

\vers{1}{
\chord{A2}Na Pradědu sněží, po zavátých šutrech\\
jedou čtyři kněží na svých sněžných skútrech;\\
s ošlehanou pletí, důstojní a vážní\\
a nad nima letí andělíčci strážní.
}

\vers{2}{
Kde končí jejich mise, to nikdo nedoví se.\\
U šikmejch očí v Číně? U šikmé věže v Pise?\\
Přísně na mě hledí, názor mají vlastní,\\
sami nejlíp vědí, kdo budou ti šťastní.
}

\vers{3}{
Budete-li v Dillí, tak mám tam jednu známou,\\
duchovně tam sílí a zná se s Dalajlámou.\\
Velikánskou pusu jí klidně dejte za mě\\
a chce-li poslat husu, tož ať napíše mámě.
}

\vers{4}{
Na Pradědu sněží, po zavátých šutrech\\
jedou čtyři kněží na svých sněžných skútrech.\\
Přísně na mě hledí, názor mají vlastní,\\
sami nejlíp vědí, kdo budou ti šťastní.
}
