\hyt{kometa}
\song{Kometa}
\ns

\vers{1}{
\chord{Am}Spatřil jsem kometu, oblohou letěla,\\
chtěl jsem jí zazpívat, ona mi zmizela.\\
\chord{Dm}Zmizela jako laň \chord{G\7}u lesa v remízku,\\
\chord{C}v očích mi zbylo jen \chord{E\7}pár žlutých penízků.
}
\ns

\vers{2}{
Penízky ukryl jsem do hlíny pod dubem,\\
až příště přiletí, my už tu nebudem.\\
My už tu nebudem, ach, pýcho marnivá,\\
spatřil jsem kometu, chtěl jsem jí zazpívat.
}
\ns

\refrainn{1}{
\chord{Am}O vodě, o trávě, \chord{Dm}o lese,\\
\chord{G\7}o smrti, se kterou smířit \chord{C}nejde se,\\
\chord{Am}o lásce, o zradě, \chord{Dm}o světě\\
\chord{E}a o všech lidech, co \chord{E\7}kdy žili na téhle \chord{Am}planetě.
}
\ns

\vers{3}{
Na hvězdném nádraží cinkají vagóny,\\
pan Kepler rozepsal nebeské zákony,\\
hledal, až nalezl v hvězdářských triedrech\\
tajemství, která teď neseme na bedrech.
}
\ns

\vers{4}{
Velká a odvěká tajemství přírody,\\
že jenom z člověka, člověk se narodí,\\
že kořen s větvemi ve strom se spojuje\\
a krev našich nadějí vesmírem putuje.
} \refsm{} Na na na\dots
\ns

\vers{5}{
Spatřil jsem kometu, byla jak reliéf,\\
zpod rukou umělce, který už nežije,\\
šplhal jsem do nebe, chtěl jsem ji osahat,\\
marnost mě vysvlékla celého donaha. 
}
\ns

\vers{6}{
Jak socha Davida z bílého mramoru\\
stál jsem a hleděl jsem, hleděl jsem nahoru,\\
až příští přiletí, ach, pýcho marnivá,\\
my už tu nebudem, ale jiný jí zazpívá.
}
\ns

\refrainn{2}{
O vodě, o trávě, o lese,\\
o smrti, se kterou smířit nejde se,\\
o lásce, o zradě, o světě,\\
bude to písnička o nás a kometě\dots
}
