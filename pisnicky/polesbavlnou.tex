\hyt{polesbavlnou}
\song{Pole s bavlnou}\interpret{rangers}{Rangers}

\vers{1}{
Pane \chord{C}můj, co na nebi je tvůj \chord{C\7}dům, má máma můj \chord{F}život dala \chord{C}katům,\\
katům \chord{C}mým v polích s bavl\chord{G\7}nou.\\
Pane \chord{C}můj, co na nebi je tvůj \chord{C\7}dům, má máma můj \chord{F}život dala \chord{C}katům,\\
katům \chord{C}mým v \chord{G\7}polích s bavl\chord{C}nou.
}

\refrain{
\chord{C\7}Den za dnem \chord{F}kůže zná bič katů,\\
vidíš \chord{C}jen černý záda bratrů,\\
jak tam \chord{C}jdou v polích s bavl\chord{G\7}nou.\\
To, co \chord{C}znáš ty v Louisi\chord{C\7}aně,\\
černý záda \chord{F}znaj' i v Texark\chord{C}aně\footnote{Texarkana je město ve východním Texasu. Je pojmenované podle své pozice na pomezí tří států - \textbf{Tex}as, \textbf{Ark}ansas a Louisi\textbf{ana}. Své dětství zde strávil Scott Joplin.},\\
i tam \chord{C}jsou v \chord{G\7}polích s bavl\chord{C}nou.
}

\vers{2}{
\rep{Já vím, že brzy musí přijít soud,\\
černý záda práva na něm vyhrajou,\\
boží soud v polích s bavlnou.}
}\refsm{}

\vers{3}{
\rep{Chtěl bych jít na potem vlhký lány,\\
tak řekni, pane můj, černý zvoň hrany,\\
katům mým v polích s bavlnou.}
}\refsm{}

\vers{4}{
\rep{Dnes měj, pane, co v nebi je tvůj dům,\\
mou duši, když život patří katům,\\
katům mým v polích s bavlnou.}
}\refsm{}
