\hyt{laskajevytah}
\song{Láska je výtah}
\hyl{saraperchetiamo}{Sar\`{a} perché ti amo}

\vers{1}{
\chord{E}Jedeme vzhůru, jsme spolu ve výtahu, \chord{C\kk m}tvářím se vážně, nebo mám aspoň snahu.\\
\chord{A}V duchu si říkám, že jsme tu jako v kleci, \chord{H}není kam couvnout, to jsou mi ale věci.
}

\vers{2}{
Tenhleten výtah, ten má už svoje roky, někdy se plouží, jindy zazs dělá skoky,\\
někdy se kývá, tak jako ta věž v Pise, teď zhaslo světlo, a výtah zastavil se.
}

\refrainn{1}{
\chord{E}Láska si dělá, co chce, krotitel nemá šanci,\\
\chord{C\kk m}láska je světlo i tma, když tma, tak jako v ranci,\\
\chord{A}s láskou je dobře i zle, teď už nám začalo \chord{H}svítat,\\
že láska je výtah, co tě veze kam chce.\chord{E}
}

\refrainn{2}{
Láska si dělá, co chce, přijde a zase zmizí, láska je ticho i třes, má svoji vlastní vizi,\\
láska je všude, i zde, může ti ubalit ránu, ač nemáš to v plánu, jenže lásce se chce.
}

\vers{3}{
Teď máme půlnoc, i když je odpoledne, i potmě cítím, že mi tvář jaksi bledne.\\
V tom tichu toužím slyšet zas něčí slova, a jako dítě svou ruku v jiné schovat.
}

\vers{4}{
Tiše ti říkám, s výrazem filosofa, chvíli se vznášet, že není katastrofa,\\
láska je výtah, co někdy staví v nebi, i když se houpá, tak je v něm škoda nebýt.
} \refsm{1}

\cod{

\chord{A}láska je všude i zde, v duchu si říkám, že by?
\chord{H}Láska je výtah, co někdy končí v nebi. \chord{E}
}
