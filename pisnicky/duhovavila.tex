\hyt{duhovavila}
\song{Duhová víla}

\vers{1}{
\chord{E}Koukej, \chord{H}kapky se \chord{E}kou\chord{H}lí, \chord{E}mám jich \chord{H}na dlani \chord{E}víc než ty. \chord{H}\emph{Jsi paní} \chord{E}\emph{vod.}\\
Já \chord{H}ptám se tě, \chord{C\kk m} jsou-\chord{A}li \chord{E}tvým dlaním \chord{H}mé kapky \chord{E}vhod? \emph{Pokud se} \chord{H}\emph{deště dotý}\chord{E}\emph{ká, já\dots}\\
\chord{H}Vím, že jsem \chord{E}hloupá, \chord{H}\emph{chtěl jsem jen} \chord{E}\emph{říct,} máš \chord{H}se mnou jen \chord{E}trápe\chord{H}ní, \emph{ty přece} \chord{E}\emph{víš,}\\
jé, \chord{H}duha nám \chord{C\kk m} \sm stou\chord{A}pá, \chord{F\kk}září, musíme blíž! \chord{H}\emph{Jsme krásně promočení!}\\
\vinv
\chord{E}Koukej, koukej, \chord{A}duhový \chord{E}sál, \chord{A}\nc\chord{G\kk m}\mm\nc\chord{H}\\
\chord{E}Koukej, koukej, \chord{A}duhový \chord{E}bál, \chord{A}\nc\chord{G\kk m}\sm\nc\chord{F\kk m}\sm\nc\chord{H}\emph{a na tom bále chybí \chord{E}král.}
}

\vers{2}{
Jsem duhová víla, \emph{A co jsem já?} v tvém pohledu slunce mám. \emph{Už nesu mrak.}\\
Jsem růžově bílá, tančím, koukej se, jak! \emph{Pokud se tance dotýká,}\\
Můj rytíři vánků, \emph{Mám tasit meč?} máš ve dlani úděl náš, \emph{Vrhnu se v seč.}\\
k tvým hořícím spánkům dávám duhovou stráž, \emph{co voní po heřmánku.}\\
\vinv
Stoupá, stoupá duhový sál,\\
houpá, houpá duhový bál, \emph{teď abych v pohádce hrál.}
}

\vers{3}{
Koukej, duha se ztrácí, \emph{To ustal déšť.} mám pojednou z lásky strach. \emph{To se ti zdá,\\
vždyť zpívají ptáci!} Tak vezmi růži a najdi jí práh. \emph{Ať vůni nepoztrácí.}\\
Já jsem duhová víla \emph{a paní vod,} mé oči jsou déšť a jas, \emph{to všechno vím,}\\
jsem růžově bílá a všechno slunce i vláha je v nás, \emph{jsi zkrátka moje milá!}\\
\vinv
Stoupá, stoupá duhový sál,\\
houpá, houpá duhový bál \emph{a zatím není to dál.}
}
