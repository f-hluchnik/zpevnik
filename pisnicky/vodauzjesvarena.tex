\hyt{vodauzjesvarena}
\song{Voda už je svařená} \interpret{michalhorak}{Michal Horák}
\note{oroginál v D moll}

\refrain{
\chord{Am}Voda už je svařená, \chord{G}svařila ji \chord{C}Mařena,\\
\chord{D}aby čajem zapila \chord{E}to, co večer \chord{Am}vypila.
}

\vers{1}{
\chord{Am}Ještě toho večera \chord{G}říkal táta \chord{C}Mařeně:\\
\uv{\chord{D}Ještě taháš kačera, \chord{E}tak se chovej zodpovědně.}\\
\vinv
Řekla na to Mařena: \uv{Neboj, já pít nebudu,\\
neudělám ostudu, jako, že jsem Mařena.}
}\refsm{}

\vers{2}{
Polovinu oslavy Mařena se nebaví,\\
to protože po chvíli byli všichni opilí.\\
\vinv
Pak přišel Karel z konzumu, v ruce flašku od rumu.\\
\uv{Ahoj, já jsem Mařena, dej sem toho Morgana!}
}\refsm{}

\vers{3}{
Vzbudila se Mařena, vedle sebe kořena,\\
v neoprénu oděná, hlava jako praštěná.\\
\vinv
Tak zmateně se na sudu od piva posadila:\\
\uv{Pít už nikdy nebudu} a vodu na čaj postavila
}\refsm{} \dots to, co večer tropila.
