\hyt{donebes}
\song{Do nebes}
\ns

\vers{1}{
\chord{Em}Neste mě, neste, ptáci, \chord{Am}do nebes,\\
\chord{H}k divokým kachnám, volám \chord{Em}přes vrátka. \chord{H}\\
\chord{Em}Skřivan ať zanechá mě \chord{Am}blízko hvězd\\
\chord{H}a straky odnesou má \chord{Em}zrcátka. \chord{H}
}

\vers{2}{
Neste mě, neste, ptáci, do nebes,\\
divokým kachnám čechrám peřinky.\\
Přes moře žluté přejdu lila vřes\\
a straky odnesou mé hodinky, ještě dnes.
}

\refrainn{1}{
\chord{C}Hou, hou, hou, \chord{Em}kam zas všichni jdou?\\
\chord{Am}Za kouskem nebe dojíš se chlebem,\\
\chord{H}a ten je s medem, \chord{H\7}dokud dva jeden jsou.
}

\vers{3}{
Neste mě, neste, ptáci, do nebes,\\
netřeba na jih, stačí Polárka.\\
Neste mě, neste, ještě voní bez,\\
tam, kde je Petřín a kde Vikárka, cigárka.
} \refsm{1}
\ns

\refrainn{2}{
Tá tada \chord{Em}da, ta tada dáta \chord{Am}tata, ta da \chord{D\7}dáta, ta da \chord{G}da. \chord{H\7}\\
Tá tada \chord{Em}da, ta tada dáta \chord{Am}tata, ta da \chord{H\7}dáta ta ta \chord{Em}ta. \chord{H\7}
}

\vers{4}{
Neste mě, neste, ptáci, do nebes,\\
a nebo stačí tam, kde Nežárka.\\
A jestli za Nežárkou vzplane les,\\
blafne i ve mně tahle stodola, všechno dokola.
}

\vers{5}{
Neste mě, neste, ptáci, do nebes,\\
divokým kachnám hážu přes vrátka.\\
Skřivan ať zanechá mě blízko hvězd,\\
až straky odnesou má zrcátka, až štěkne pes.
} \refsm{1} \refsm{2} \note{fade out}
